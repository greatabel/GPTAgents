# 把“外部服务的连接配置”注册到 promptflow 里
simple_flow 目录下
 pf connection create \
  --file openai.yaml \
  --name local_openai


# 运行promptflow 测试
pf flow test \
  --flow . \
  --inputs user_input="推荐一些东西"

或者：
pf flow test --flow .

对于recommender_with_grounding工程：
pf run create --flow . --data bulk_recommend.jsonl



# pf flow serve（本地 HTTP 服务）
pf flow serve --source . --port 8080

测试
curl -s -X POST "http://localhost:8080/score" \
  -H "Content-Type: application/json; charset=utf-8" \
  -d '{
    "user_input": "推荐几部时间旅行电影"
  }' | jq

pf run create \
